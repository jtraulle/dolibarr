33 function dolStripPhpCode($str, $replacewith = '')
34 {
35  $newstr = '';
36 
37  //split on each opening tag
38  $parts = explode('<?php', $str);
39  if (!empty($parts))
40  {
41  $i=0;
42  foreach($parts as $part)
43  {
44  if ($i == 0) // The first part is never php code
45  {
46  $i++;
47  $newstr .= $part;
48  continue;
49  }
50  // The second part is the php code. We split on closing tag
51  $partlings = explode('?>', $part);
52  if (!empty($partlings))
53  {
54  $phppart = $partlings[0];
55  //remove content before closing tag
56  if (count($partlings) > 1) $partlings[0] = ''; // Todo why a count > 1 and not >= 1 ?
57  //append to out string
58  $newstr .= '<span class="phptag">'.$replacewith.'<!-- '.$phppart.' --></span>'.implode('', $partlings);
59  }
60  }
61  }
62  return $newstr;
63 }
