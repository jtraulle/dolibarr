451  private function loadLotStock($productIdArray = array())
452  {
453  global $conf, $langs;
454 
455  $cacheLoaded = false;
456  if (empty($productIdArray))

458  // only Load lot stock for given products

461  }
462  if (count($productIdArray) && count($this->cache_lot))
463  {
464  // check cache already loaded for product id's

467  $cacheLoaded = ! empty($this->cache_lot[$productId]) ? true : false;
468  }
469  }
470  if ($cacheLoaded)
471  {
472  return count($this->cache_lot);
473  }
474  else
475  {
476  // clear cache
477  $this->cache_lot = array();
