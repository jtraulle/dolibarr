132  public function getListOfCountries($sortfield = "code", $sortorder = 'ASC', $limit = 100, $page = 0, $filter = '', $lang = '', $sqlfilters = '')

143  if (! DolibarrApi::_checkFilters($sqlfilters))
144  {
145  throw new RestException(503, 'Error when validating parameter sqlfilters '.$sqlfilters);
146  }
147  $regexstring='\(([^:\'\(\)]+:[^:\'\(\)]+:[^:\(\)]+)\)';
148  $sql.=" AND (".preg_replace_callback('/'.$regexstring.'/', 'DolibarrApi::_forge_criteria_callback', $sqlfilters).")";
149  }
150 
151  $sql.= $this->db->order($sortfield, $sortorder);
152 
153  if ($limit) {
154  if ($page < 0) {
155  $page = 0;
156  }
157  $offset = $limit * $page;
158 
159  $sql .= $this->db->plimit($limit, $offset);
160  }
161 
162  $result = $this->db->query($sql);
163 
164  if ($result) {
165  $num = $this->db->num_rows($result);
166  $min = min($num, ($limit <= 0 ? $num : $limit));
167  for ($i = 0; $i < $min; $i++) {
168  $obj = $this->db->fetch_object($result);
169  $country = new Ccountry($this->db);
170  if ($country->fetch($obj->rowid) > 0) {
171  // Translate the name of the country if needed
172  // and then apply the filter if there is one.
173  $this->translateLabel($country, $lang);
174 
175  if (empty($filter) || stripos($country->label, $filter) !== false) {
176  $list[] = $this->_cleanObjectDatas($country);
177  }
178  }
179  }
180  } else {
181  throw new RestException(503, 'Error when retrieving list of countries : '.$country->error);
182  }
183

303  private function translateLabel($country, $lang)
304  {
305  if (!empty($lang)) {
306  // Load the translations if this is a new language.
307  if ($this->translations == null || $this->translations->getDefaultLang() !== $lang) {
308  global $conf;
309  $this->translations = new Translate('', $conf);
310  $this->translations->setDefaultLang($lang);
311  $this->translations->load('dict');
312  }
313  if ($country->code) {
314  $key = 'Country'.$country->code;
315  $translation = $this->translations->trans($key);
316  if ($translation != $key) {
317  $country->label = html_entity_decode($translation);
318  }
319  }
320  }
