164  // La requete doit retourner: id, email, fk_contact, lastname, firstname

166  $sql.= " u.lastname, u.firstname as firstname, u.civility as civility_id, u.login, u.office_phone";
167  $sql.= " FROM ".MAIN_DB_PREFIX."user as u";
168  $sql.= " WHERE u.email <> ''"; // u.email IS NOT NULL est implicite dans ce test
169  $sql.= " AND u.entity IN (0,".$conf->entity.")";
170  $sql.= " AND u.email NOT IN (SELECT email FROM ".MAIN_DB_PREFIX."mailing_cibles WHERE fk_mailing=".$mailing_id.")";
171  if (isset($_POST["filter"]) && $_POST["filter"] == '1') $sql.= " AND u.statut=1";
172  if (isset($_POST["filter"]) && $_POST["filter"] == '0') $sql.= " AND u.statut=0";
173  if (isset($_POST["filteremployee"]) && $_POST["filteremployee"] == '1') $sql.= " AND u.employee=1";
174  if (isset($_POST["filteremployee"]) && $_POST["filteremployee"] == '0') $sql.= " AND u.employee=0";
175  $sql.= " ORDER BY u.email";
176 
177  // Stocke destinataires dans cibles
178  $result=$this->db->query($sql);
179  if ($result)
180  {
181  $num = $this->db->num_rows($result);
182  $i = 0;
183  $j = 0;
184 
185  dol_syslog(get_class($this)."::add_to_target mailing ".$num." targets found");
186 
187  $old = '';
188  while ($i < $num)
189  {
190  $obj = $this->db->fetch_object($result);
191  if ($old <> $obj->email)
192  {
193  $cibles[$j] = array(
194  'email' => $obj->email,
195  'fk_contact' => $obj->fk_contact,
196  'lastname' => $obj->lastname,
197  'firstname' => $obj->firstname,
198  'other' =>

200  ($langs->transnoentities("UserTitle").'='.$obj->civility_id).';'.
201  ($langs->transnoentities("PhonePro").'='.$obj->office_phone),
202  'source_url' => $this->url($obj->id),
203  'source_id' => $obj->id,
204  'source_type' => 'user'
205  );
206  $old = $obj->email;

208  }
