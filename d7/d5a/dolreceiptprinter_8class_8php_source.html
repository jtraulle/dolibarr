650  public function initPrinter($printerid)
651  {
652  global $conf;
653  $error=0;
654  $sql = 'SELECT rowid, name, fk_type, fk_profile, parameter';
655  $sql.= ' FROM '.MAIN_DB_PREFIX.'printer_receipt';
656  $sql.= ' WHERE rowid = '.$printerid;
657  $sql.= ' AND entity = '.$conf->entity;
658  $resql = $this->db->query($sql);
659  if ($resql) {
660  $obj = $this->db->fetch_array($resql);
661  } else {
662  $error++;
663  $this->errors[] = $this->db->lasterror;
664  }
665  if (empty($obj)) {
666  $error++;
667  $this->errors[] = 'PrinterDontExist';
668  }
669  if (! $error) {
670  $parameter = $obj['parameter'];
671  try {
672  switch ($obj['fk_type']) {
673  case 1:
674  require_once DOL_DOCUMENT_ROOT .'/includes/mike42/escpos-php/src/DummyPrintConnector.php';
675  $this->connector = new DummyPrintConnector();
676  break;
677  case 2:
678  $this->connector = new FilePrintConnector($parameter);
679  break;
