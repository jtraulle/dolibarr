95  public function index($sortfield = "t.rowid", $sortorder = 'ASC', $limit = 100, $page = 0, $typeid = '', $sqlfilters = '')
96  {
97  global $db, $conf;
98 
99  $obj_ret = array();
100 
101  if(! DolibarrApiAccess::$user->rights->adherent->lire) {
102  throw new RestException(401);
103  }
104 
105  $sql = "SELECT t.rowid";
106  $sql.= " FROM ".MAIN_DB_PREFIX."adherent as t";
107  $sql.= ' WHERE t.entity IN ('.getEntity('adherent').')';
108  if (!empty($typeid))
109  {
110  $sql.= ' AND t.fk_adherent_type='.$typeid;
111  }
112  // Add sql filters
113  if ($sqlfilters)
114  {
115  if (! DolibarrApi::_checkFilters($sqlfilters))
116  {
117  throw new RestException(503, 'Error when validating parameter sqlfilters '.$sqlfilters);
118  }
119  $regexstring='\(([^:\'\(\)]+:[^:\'\(\)]+:[^:\(\)]+)\)';
120  $sql.=" AND (".preg_replace_callback('/'.$regexstring.'/', 'DolibarrApi::_forge_criteria_callback', $sqlfilters).")";
121  }
122 
123  $sql.= $db->order($sortfield, $sortorder);
124  if ($limit) {
125  if ($page < 0)
126  {
127  $page = 0;
128  }
129  $offset = $limit * $page;
130 
131  $sql.= $db->plimit($limit + 1, $offset);
132  }
133 
134  $result = $db->query($sql);
135  if ($result)
136  {
137  $i=0;
138  $num = $db->num_rows($result);
139  $min = min($num, ($limit <= 0 ? $num : $limit));
140  while ($i < $min)
141  {
142  $obj = $db->fetch_object($result);
143  $member = new Adherent($this->db);
144  if($member->fetch($obj->rowid)) {
145  $obj_ret[] = $this->_cleanObjectDatas($member);
146  }
147  $i++;
148  }
149  }
150  else {
151  throw new RestException(503, 'Error when retrieve member list : '.$db->lasterror());
152  }
153  if( ! count($obj_ret)) {
154  throw new RestException(404, 'No member found');
155  }
156 
157  return $obj_ret;
158  }

191  public function put($id, $request_data = null)
192  {
193  if(! DolibarrApiAccess::$user->rights->adherent->creer) {
194  throw new RestException(401);
195  }
196 
197  $member = new Adherent($this->db);
198  $result = $member->fetch($id);
199  if( ! $result ) {
200  throw new RestException(404, 'member not found');
201  }
202 
203  if( ! DolibarrApi::_checkAccessToResource('member', $member->id)) {
204  throw new RestException(401, 'Access not allowed for login '.DolibarrApiAccess::$user->login);
205  }
206 
207  foreach($request_data as $field => $value) {
208  if ($field == 'id') continue;
209  // Process the status separately because it must be updated using
210  // the validate() and resiliate() methods of the class Adherent.
211  if ($field == 'statut') {
212  if ($value == '0') {
213  $result = $member->resiliate(DolibarrApiAccess::$user);
214  if ($result < 0) {
215  throw new RestException(500, 'Error when resiliating member: '.$member->error);
216  }
217  } elseif ($value == '1') {
218  $result = $member->validate(DolibarrApiAccess::$user);
219  if ($result < 0) {
220  throw new RestException(500, 'Error when validating member: '.$member->error);
221  }
222  }
223  } else {
224  $member->$field = $value;
225  }
226  }
227 
228  // If there is no error, update() returns the number of affected rows
229  // so if the update is a no op, the return value is zero.
230  if ($member->update(DolibarrApiAccess::$user) >= 0)
231  {
232  return $this->get($id);
233  }
234  else
235  {
236  throw new RestException(500, $member->error);
237  }
238  }
