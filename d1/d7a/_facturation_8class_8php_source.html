95  public function ajoutArticle()
96  {
97  global $conf,$db,$mysoc;
98 
99  $thirdpartyid = $_SESSION['CASHDESK_ID_THIRDPARTY'];
100 
101  $societe = new Societe($db);
102  $societe->fetch($thirdpartyid);
103 
104  $product = new Product($db);
105  $product->fetch($this->id);
106 
107 
108  $vatrowid = $this->tva();
109 
110  $tmp = getTaxesFromId($vatrowid);
111  $txtva = $tmp['rate'].(empty($tmp['code'])?'':' ('.$tmp['code'].')');
112  $vat_npr = $tmp['npr'];
113 
114  $localtaxarray = getLocalTaxesFromRate($vatrowid, 0, $societe, $mysoc, 1);
115 
116  // Clean vat code
117  $vat_src_code='';
118  if (preg_match('/\((.*)\)/', $txtva, $reg))
119  {
120  $vat_src_code = $reg[1];
121  $txtva = preg_replace('/\s*\(.*\)/', '', $txtva); // Remove code into vatrate.
122  }
123 
124  // Define part of HT, VAT, TTC
125  $resultarray=calcul_price_total($this->qte, $this->prix(), $this->remisePercent(), $txtva, -1, -1, 0, 'HT', $vat_npr, $product->type, $mysoc, $localtaxarray);
126 
127  // Calculation of total HT without discount
128  $total_ht = $resultarray[0];
129  $total_vat = $resultarray[1];
130  $total_ttc = $resultarray[2];
131  $total_localtax1 = $resultarray[9];
132  $total_localtax2 = $resultarray[10];
133 
134  // Calculation of the discount amount
135  if ($this->remisePercent())
136  {
137  $remise_percent = $this->remisePercent();
138  } else {
139  $remise_percent = 0;
140  }
141  $montant_remise_ht = ($resultarray[6] - $resultarray[0]);
142  $this->montantRemise($montant_remise_ht);
143 
144  $newcartarray=$_SESSION['poscart'];
145 
146  $i = 0;
147  if (!is_null($newcartarray) && !empty($newcartarray)) {
148  $i=count($newcartarray);
149  }
150 
151  $newcartarray[$i]['id']=$i;
152  $newcartarray[$i]['ref']=$product->ref;
153  $newcartarray[$i]['label']=$product->label;
154  $newcartarray[$i]['price']=$product->price;
155  $newcartarray[$i]['price_ttc']=$product->price_ttc;
156 
157  if (! empty($conf->global->PRODUIT_MULTIPRICES))
158  {
159  if (isset($product->multiprices[$societe->price_level]))
160  {
161  $newcartarray[$i]['price'] = $product->multiprices[$societe->price_level];
162  $newcartarray[$i]['price_ttc'] = $product->multiprices_ttc[$societe->price_level];
163  }
164  }
165 
166  $newcartarray[$i]['fk_article']=$this->id;
167  $newcartarray[$i]['qte']=$this->qte();
168  $newcartarray[$i]['fk_tva']=$this->tva(); // Vat rowid
169  $newcartarray[$i]['remise_percent']=$remise_percent;
170  $newcartarray[$i]['remise']=price2num($montant_remise_ht);
171  $newcartarray[$i]['total_ht']=price2num($total_ht, 'MT');
172  $newcartarray[$i]['total_ttc']=price2num($total_ttc, 'MT');
173  $newcartarray[$i]['total_vat']=price2num($total_vat, 'MT');
174  $newcartarray[$i]['total_localtax1']=price2num($total_localtax1, 'MT');
175  $newcartarray[$i]['total_localtax2']=price2num($total_localtax2, 'MT');
176  $_SESSION['poscart']=$newcartarray;
177 
178  $this->raz();
179  }
